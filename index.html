const probeFiles = {};
const probeFilesPath = './probeFiles/';

// Read all probeFiles spec files into `probeFiles`
fs.readdirSync(probeFilesPath).forEach(file => {
  if (file.split(/\./).pop() === 'toml') {
    const f = toml(probeFilesPath + file);
    probeFiles[file] = f;
  } else {
    appLogger.warn(file + ' is not a toml file and are not allowed in probeFiles/ directory');
  }

const schema = schemaProbeFile.probeFile;

for (const i in probeFiles) {
  if (probeFiles[i]['Check_Type'] === 1) {
    schema.Check = schemaProbeFile.check1;
  } else if (probeFiles[i]['Check_Type'] === 2) {
    schema.Check = schemaProbeFile.check2;
  } else if (probeFiles[i]['Check_Type'] === 3) {
    schema.Check = schemaProbeFile.check3;
  } else if (probeFiles[i]['Check_Type'] === 4) {
    schema.Check = schemaProbeFile.check4;
    if (!isValidHostname(probeFiles[i]['Check']['Host'])) {
      appLogger.error(`Hostname in ${i} is ${probeFiles[i]['Check']['Host']} which is not a valid hostname`);
      process.exit();
    };
  } else {
    appLogger.error(`${probeFiles[i]} The value in Check_Type is ${probeFiles[i]['Check_Type']} which is out of range`);
  };
  const s = new Validate(schema, { strict: true });
  try {
    s.assert(probeFiles[i]);
  } catch (e) {
    appLogger.error(`${e.message} in ${i}`);
    process.exit(1);
  };

  probeFiles[i].Check_Interval = duration(probeFiles[i].Check_Interval);
  probeFiles[i].Error_Threshold.Within = duration(probeFiles[i].Error_Threshold.Within);
  probeFiles[i].Cool_Down = duration(probeFiles[i].Cool_Down);

  if (probeFiles[i].Error_Threshold.Errors === 1 && probeFiles[i].Error_Threshold.Within !== 0) {
    appLogger.error(`Error_Threshold.Within must be 0s, when Error_Threshold.Errors is 1. ${i}`);
    process.exit(1);
  };
  if (probeFiles[i].Error_Threshold.Errors < 1) {
    appLogger.error(`Error_Threshold.Errors must be equal or larger than 1. ${i}`);
    process.exit(1);
  };
  if (probeFiles[i].Check_Interval >= probeFiles[i].Cool_Down && probeFiles[i].Cool_Down !== 0) {
    appLogger.error(`Cool_Down must be larger than Check_Interval. ${i}`);
    process.exit(1);
  };
  if (probeFiles[i].Check_Interval >= probeFiles[i].Error_Threshold.Within && probeFiles[i].Error_Threshold.Within !== 0) {
    appLogger.error(`Error_Threshold.Within must be larger than Check_Interval. ${i}`);
    process.exit(1);
  };
  if (probeFiles[i].Check_Interval <= probeFiles[i].Check.Timeout) {
    appLogger.error(`Check_Interval must be larger than Check.Timeout. ${i}`);
    process.exit(1);
  };
  if (probeFiles[i].Error_Threshold.Within >= probeFiles[i].Cool_Down && !(probeFiles[i].Error_Threshold.Within === 0 || probeFiles[i].Cool_Down === 0)) {
    appLogger.error(`Cool_Down must be larger than Error_Threshold.Within. ${i}`);
    process.exit(1);
  };
  if (probeFiles[i].Cool_Down % probeFiles[i].Error_Threshold.Within !== 0 && probeFiles[i].Error_Threshold.Within !== 0) {
    appLogger.error(`Cool_Down/Error_Threshold.Within must be an integer. ${i}`);
    process.exit(1);
  };
  if (probeFiles[i].Error_Threshold.Within % probeFiles[i].Check_Interval !== 0) {
    appLogger.error(`Error_Threshold.Within/Check_Interval must be an integer. ${i}`);
    process.exit(1);
  };
  if (probeFiles[i].Cool_Down % probeFiles[i].Check_Interval !== 0) {
    appLogger.error(`Cool_Down/Check_Interval must be an integer. ${i}`);
    process.exit(1);
  };
};

const systemConfigs = require('./getSystemConfigs');

module.exports = (probeFiles) => {
  const sysC = [];
  for (const i in systemConfigs) {
    if (sysC.includes(i)) {
      appLogger.error(`Multiple system config files has system: ${i}`);
      process.exit();
    } else {
      sysC.push(i);
    };
  };

  const sysP = [];
  for (const i in probeFiles) {
    if (!sysP.includes(probeFiles[i].System)) {
      sysP.push(probeFiles[i].System);
    };
  };

  for (const i in sysP) {
    if (!sysC.includes(sysP[i])) {
      appLogger.error(`Atleast one probefile has system ${sysP[i]}. There is not a matching system config.`);
      process.exit();
    };
  };

  for (const i in sysC) {
    if (!sysP.includes(sysC[i])) {
      appLogger.warn(`A system config for system ${sysC[i]} exists, but no probefiles uses it.`);
    };
  };
};

Performance optimering

const delay = require('delay');

const { appLogger, monitorLogger } = require('./logger');
const Clock = require('./clock');
const selectProbe = require('./selectProbe');
const jiraIncident = require('./jiraIncident');

module.exports = async (probeFile, checkStates) => {
  const timer = new Clock(probeFile.Check_Interval);
  timer.start();
  const coolDownTimer = new Clock(probeFile.Check_Interval);
  const globalTimer = new Clock(probeFile.Check_Interval);
  globalTimer.start();

  let ret;
  let errorsSeen = 0;
  let timePassed = 0;

  const filename = probeFile['Filename'];
  const eksternprobe = selectProbe(probeFile);

  while (true) {
    ret = await eksternprobe(probeFile);

    if (ret.ok <= 0) {
      monitorLogger.info('Down', { probeFile: filename, state: 'Logic later on will determine weather an incident should be created' });
      errorsSeen++;

      if (errorsSeen >= probeFile.Error_Threshold.Errors && timePassed <= probeFile.Error_Threshold.Within
          || probeFile.Error_Threshold.Within === 0) {
        errorsSeen = 0;
        timePassed = 0;
        timer.reset();

        if (coolDownTimer.read() === 0 || coolDownTimer.read() >= probeFile.Cool_Down) {
          jiraIncident(probeFile, ret).then(response => {
            monitorLogger.info('Down', { probeFile: filename, state: `Created Jira incident ${response.message} sucessfully` });

            if (probeFile['Expose_State']) {
              checkStates[filename] = 'Down';
              monitorLogger.info('Down', { probeFile: filename, state: 'Updated JSON to Down' });
            };
          }).catch(reject => {
            appLogger.error(`Could not create Jira incident for ${filename}. Reject: ${reject.message}`);
          });

          globalTimer.reset();
          if (coolDownTimer.read() === 0 && probeFile.Cool_Down > 0) {
            coolDownTimer.start();
          } else if (coolDownTimer.read() >= probeFile.Cool_Down) {
            coolDownTimer.reset();
          };
        };
      } else if (timePassed >= probeFile.Error_Threshold.Within
                 && probeFile.Error_Threshold.Within !== 0) {
        appLogger.warn('Time threshold exceeded. Resetting', { Error_Threshold_Within: probeFile.Error_Threshold.Within, probeFile: filename });
        errorsSeen = 0;
        timePassed = 0;
      } else {
        monitorLogger.info('Down', { probeFile: filename, state: 'Threshold not meet yet. Everything is still fine' });
      };
    } else {
      if (probeFile['Expose_State']) {
        checkStates[filename] = 'Up';
      };
      monitorLogger.info('Up', { probeFile: filename, state: 'Everything is fine' });

      if (probeFile['Expose_State'] && checkStates[filename] === 'Down') {
        checkStates[filename] = 'Up';
        monitorLogger.info('Up', { probeFile: filename, state: 'Updated JSON to Up' });
      };
    };

    await delay(probeFile.Check_Interval);
    timePassed += probeFile.Check_Interval;
  };
};

const getStatusCode = require('./getStatusCode');

const getSecureString = require('./getSecureString');

module.exports = (probeFile) => {
  function selectFunction(probeFile) {
    switch (probeFile['Check_Type']) {
      case 1:
        return getStatusCode;
        break;
      case 2:
        return getString;
        break;
      case 3:
        return getSecureString;
        break;
      case 4:
        return getHttpsCertificate;
        break;
      default:
        throw new Error(`Ikke gyldig Check_Type i probefil. ${probeFile['Check_Type']} in ${probeFile['Filename']}`);
    };
  };
  // returns a reference to one of the get-functions
  return selectFunction(probeFile);
};

  } else {

const config = require('./getConfig');

module.exports = function deleteDupsAndUpdateDocStore(wh, udvSearchString, docStore) {
  if (udvSearchString === config.catch_all_search_string) {
    // delete/modify catch all webhook
    // remove duplicate docs
    const docsWithoutDups = [];
    wh.documents.forEach((k) => {
      // search for _id's in docStore
      const id = k._id;
      if (docStore[wh.system][id] === undefined) {
        docsWithoutDups.push(k);
      };
    });

    wh.documents = docsWithoutDups;
    wh.matched_documents = docsWithoutDups.length;

    // Clear docStore because a "catch all" webhook have been received
    // it is therefore strongly recommended that each system only have ONE catch all.
    docStore[wh.system] = {};
  } else {
    // webhook is from a regular Alertfile
    // add new docs to docStore
    wh.documents.forEach((k) => {
      const id = k._id;
      if (docStore[wh.system][id] === undefined) {
        docStore[wh.system][id] = k;
      };
    });
  };

  return { wh, docStore };

import { intervalToDuration, getDaysInMonth, format, isWithinInterval, parseISO } from 'date-fns';
import * as _ from 'lodash';
import { convert } from 'cashify-es2017';
import Big from 'big.js';
import { strict as assert } from 'assert';
import { validateObject } from './validateObject';

import { readToml } from './readToml';

const nt = readToml('./config/nat.toml');
const r = readToml('./config/currency.toml');

export function expandRedhatOffer(
  mpmDb: Record<string, any>,
  products: Record<string, any>,
  customerNumber: number,
  offerId: number,
  today: Date,
  natTable: Record<string, any> = nt,
  rates: Record<string, number> = r,
) {
  assert(typeof mpmDb === 'object');
  assert(typeof products === 'object');
  assert(typeof customerNumber === 'number');
  assert(typeof offerId === 'number');
  assert(typeof today === 'object');
  assert(typeof natTable === 'object');
  assert(typeof rates === 'object');

  validateObject(mpmDb, customerNumber, 'offers', offerId);
  const offerExpanded = _.cloneDeep(mpmDb[customerNumber].offers[offerId]);

  type Redhat = {
    productType: string;
    productNo: string;
    dateStart: Date;
    dateEnd: Date;
    specialPrice: number;
    producerDiscount: number;
    producerDiscountCustomer: number;
    miracleDiscount: number;
    listPrice: number;
    amount: number;
    calcPrice: number;
    description: string;
  };

  const pKey = getProductKey(products, today);

  let hasSpecialPrice = false;
  let hasCurrency = '';
  offerExpanded.products.forEach((p: Redhat) => {
    switch (p.productType) {
      case 'redhat':
        const startDate = new Date(p.dateStart);
        const endDate = new Date(p.dateEnd);

        const duration = intervalToDuration({ start: startDate, end: endDate });
        const years = duration.years as number;

        let msrpType = 0;
        if (years < 1) {
          if (p.productNo.endsWith('F3')) {
            throw Error('Der kan ikke laves tilbud på subscriptions på under et år med en 3års subscription');
          }
          console.log('Det indtastede tidsinterval er mindre end et år');
          msrpType = 1;
        } else if (years >= 1 && years < 3) {
          if (p.productNo.endsWith('F3')) {
            throw Error('Der kan ikke laves tilbud på subscriptions på under 3 år med en 3års subscription');
          }
          msrpType = 1;
        } else if (years >= 3) {
          if (!p.productNo.endsWith('F3')) {
            throw Error('Der kan ikke laves tilbud på subscriptions på over 3 år med en 1års subscription');
          }
          msrpType = 3;
        } else {
          throw Error('Der er noget galt med antallet af år');
        }

        const msrpName = `${msrpType}_year_msrp`;
        validateObject(products, pKey, p.productNo, msrpName);
        p.listPrice = products[pKey][p.productNo][msrpName] as number;
        hasCurrency = setCurrency(hasCurrency, products[pKey][p.productNo].currency);

        let price = p.listPrice;
        if (p.specialPrice) {
          price = p.specialPrice;
          hasSpecialPrice = true;
        }

        let producerDiscount = p.producerDiscount;
        if (p.producerDiscountCustomer) producerDiscount = p.producerDiscountCustomer;

        let miracleDiscount = 0 as number;
        if (p.miracleDiscount !== null) miracleDiscount = p.miracleDiscount;

        const months = duration.months as number;
        const days = duration.days as number;
        const daysInEndDateMonth = getDaysInMonth(endDate);

        const sellingTerm = years + months / 12 + (days + 1) / daysInEndDateMonth / 12 / msrpType;

        p.calcPrice = price * p.amount * sellingTerm * (1 - (producerDiscount + miracleDiscount) / 100);

        p.description = products[pKey][p.productNo].sku_description;
        break;
      case 'text':
        break;
      default:
        throw Error('problem: unknown product type');
    }
  });

  offerExpanded.totalPrice = 0 as number;
  offerExpanded.products.forEach((p: Redhat) => {
    if (p.calcPrice) {
      offerExpanded.totalPrice = offerExpanded.totalPrice + p.calcPrice;
    }
  });

  offerExpanded.nat_discount = getNatDiscount(natTable, offerExpanded.totalPrice, offerExpanded.currency);

  if (hasSpecialPrice && !offerExpanded.nat_discount_customer) offerExpanded.nat_discount_customer = 0;

  offerExpanded.totalPrice *=
    1 - (offerExpanded.nat_discount_customer + offerExpanded.deal_reg_discount_customer) / 100;

  if (hasCurrency !== offerExpanded.currency) {
    offerExpanded.currency_rate = rates[hasCurrency];

    offerExpanded.totalPrice = convert(offerExpanded.totalPrice, {
      from: hasCurrency,
      to: offerExpanded.currency,
      BigJs: Big,
      base: 'EUR',
      rates,
    });
  }

  offerExpanded.today = format(new Date(), 'dd MMMM yyyy');

  offerExpanded.electronic_delivery = [];

  Object.values(mpmDb[customerNumber].contact).forEach((e: any) => {
    if (e.electronic_delivery) {
      const s = `${e.firstname} ${e.lastname} (${e.email})`;
      offerExpanded.electronic_delivery.push(s);
    }
  });

  delete offerExpanded.customer_address;
  delete offerExpanded.customer_contact;

  const adr = mpmDb[customerNumber].offers[offerId].customer_address;
  const ctt = mpmDb[customerNumber].offers[offerId].customer_contact;

  const obj = {
    customer_name: mpmDb[customerNumber].address[adr].name,
    customer_address: mpmDb[customerNumber].address[adr].address1,
    customer_zip: mpmDb[customerNumber].address[adr].postcode,
    customer_city: mpmDb[customerNumber].address[adr].city,
    customer_contact: mpmDb[customerNumber].contact[ctt].firstname + ' ' + mpmDb[customerNumber].contact[ctt].lastname,
    customer_cvr: mpmDb[customerNumber].vat_no,
  };

  mpmDb[customerNumber].offersExpanded[offerId] = {
    ...obj,
    ...offerExpanded,
  };
}

function getNatDiscount(natTable: any, totalPrice: any, currency: string) {
  for (const v of natTable[currency]) {
    if (totalPrice >= v.from && totalPrice < v.to) return v.discount;
  }
}

function setCurrency(offerCurrency: string, currency: string) {
  if (offerCurrency === '') {
    offerCurrency = currency;
    return offerCurrency;
  }
  if (offerCurrency !== currency) {
    throw Error('Mixed currencies in one offer now allowed.');
  }
  return offerCurrency;
}

function getProductKey(rhProducts: object, today: Date) {
  let r;
  for (const k of Object.keys(rhProducts)) {
    const t = k.split('--');
    const interval = { start: parseISO(t[1]), end: parseISO(t[0]) };

    if (isWithinInterval(today, interval)) {
      if (r) {
        throw Error('ERROR: multiple product lists match.');
      }
      r = k;
    }
  }

  if (r === undefined) {
    throw Error('ERROR: No list found that includes todays date');
  }

  return r;
}

      body: JSON.stringify({
